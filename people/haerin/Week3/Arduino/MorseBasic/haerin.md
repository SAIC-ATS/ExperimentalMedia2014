void dot()
{

}

void dash() 
{
  digitalWrite(led, HIGH); 
  delay(3000); 
  digitalWrite(led, LOW); 
  delay(1000);  //space between parts of same letter 
} 

void space() //space between letters
{
  digitalWrite(led, LOW); 
  delay(2000); 
} 

void pause() //space between words
{
  digitalWrite(led, LOW); 
  delay(5000); 
}
